    public float moveSpeed = 125;
    Vector3 localRotation = Vector3.zero;

    private void Update()
    {
        if (photonView.IsMine)
        {
#if UNITY_EDITOR
            //Move Front/Back
            if (Input.GetKey(KeyCode.W))
            {
                transform.Translate(transform.forward * Time.deltaTime * gameController.speed, Space.World);
            }
            else if (Input.GetKey(KeyCode.S))
            {
                transform.Translate(-transform.forward * Time.deltaTime * gameController.speed, Space.World);
            }

            //Rotate Left/Right
            if (Input.GetKey(KeyCode.A))
            {
                transform.Rotate(new Vector3(0, -15, 0) * Time.deltaTime * gameController.speed, Space.Self);
            }
            else if (Input.GetKey(KeyCode.D))
            {
                transform.Rotate(new Vector3(0, 15, 0) * Time.deltaTime * gameController.speed, Space.Self);
            }
#else
            // Movement
            transform.Translate(transform.forward * Time.deltaTime * gameController.speed, Space.World);
            // Rotation
            localRotation.y += Input.acceleration.x * Time.deltaTime * moveSpeed;
            localRotation.x -= Input.acceleration.z * Time.deltaTime * moveSpeed;
            if (localRotation.x > 10)
            {
                localRotation.x = 10;
            }
            if (localRotation.x < -40)
            {
                localRotation.x = -40;
            }
            localRotation.z = 0;
            transform.localEulerAngles = localRotation;
#endif
        }
    }
